from django.db import models

class Service(models.Model):
    SERVICE_TYPES = [
        ('cleaning services', 'Cleaning Services'),
        ('fumigation services', 'Fumigation Services'),
        ('laundry services', 'Laundry Services'),
    ]

    service_type = models.CharField(max_length=50, choices=SERVICE_TYPES)
    name = models.CharField(max_length=100) # e.g., "Standard Cleaning", "Deep Fumigation"
    description = models.TextField()
    base_price = models.DecimalField(max_digits=10, decimal_places=2)
    is_active = models.BooleanField(default=True)

    def __str__(self):
        return f"{self.name} - ({self.base_price})"

class ServicePricing(models.Model):
    """Model to define pricing options for different service types."""
    service = models.ForeignKey(Service, on_delete=models.CASCADE, related_name='service_options')
    property_type = models.CharField(max_length=50)  # e.g., "Small Apartment", "Large House", "Commercial Space"
    property_size = models.CharField(max_length=50)  # e.g., "Small", "Medium", "Large"
    base_price = models.DecimalField(max_digits=10, decimal_places=2)  
    price_multiplier = models.DecimalField(max_digits=4, decimal_places=2, default=1.0)
    description = models.TextField()

    def __str__(self):
        return f"{self.service.name} - Multiplier: {self.price_multiplier}"

class PropertySizeOption(models.Model):
    """Model to define property size options with pricing multipliers."""

        ('medium', 'Medium (3-4 rooms)'),

class AddOnType(models.TextChoices):
    DEEP_CLEANING = 'deep_cleaning', 'Deep Cleaning (+₦5,000)'
    WINDOW_CLEANING = 'window_cleaning', 'Window Cleaning (+₦3,000)'
    CARPET_CLEANING = 'carpet_cleaning', 'Carpet Cleaning (+₦2,000)'
    EMERGENCY = 'emergency', 'Emergency Service (24/7) (+₦10,000)'

class AdditionalService(models.Model):
    """Stores extra services (e.g., Deep Cleaning, Window Cleaning)."""
    addon_type = models.CharField(max_length=30, choices=AddOnType.choices, unique=True)
    price = models.DecimalField(max_digits=10, decimal_places=2, default=10000.00)
    description = models.TextField(blank=True, null=True)
    is_active = models.BooleanField(default=True)
    name = models.CharField(max_length=100, blank=True, null=True)

    @staticmethod # run this once to populate database table
    def seed_defaults(): 
        defaults = {
            AddOnType.DEEP_CLEANING: 5000.00,
            AddOnType.WINDOW_CLEANING: 3000.00,
            AddOnType.CARPET_CLEANING: 2000.00,
            AddOnType.EMERGENCY: 10000.00,
        }
        for addon, price in defaults.items():
            AdditionalService.objects.get_or_create(addon_type=addon, defaults={'price': price})

    def __str__(self):
        return f"{self.get_addon_type_display()}"

class TimeSlot(models.Model):
    start_time = models.TimeField()
    end_time = models.TimeField()
    display_name = models.CharField(max_length=50)  # e.g., "8:00 AM", "10:00 AM"
    is_active = models.BooleanField(default=True)

    # seed this once to populate database table # run with TimeSlot.seed_slots()
    @staticmethod
    def seed_slots():
        slots = [
            ("08:00", "10:00"),
            ("10:00", "12:00"),
            ("14:00", "16:00"),
            ("16:00", "18:00"),
        ]
        for start, end in slots: TimeSlot.objects.get_or_create(
                start_time=start,
                defaults={'end_time': end}
            )

    class Meta:
        unique_together = ('start_time', 'end_time')
        ordering = ['start_time']

    def __str__(self):
        return f"{self.start_time.strftime('%H:%M')} - {self.end_time.strftime('%H:%M')}"
    
class Schedule(models.Model):
    """Model to hold available scheduling dates for services."""
    date = models.DateField()
    slot = models.ForeignKey(TimeSlot, on_delete=models.CASCADE)
    is_available = models.BooleanField(default=True)

    class Meta:
        unique_together = ('date', 'slot')
        ordering = ['date', 'slot__start_time']

    def __str__(self):
        return f"{self.date} - {self.slot}"

# Main booking Model
class Booking(models.Model):
    """The main model to hold a customer's confirmed booking."""

    # service details
    main_service = models.ForeignKey(Service, on_delete=models.SET_NULL, null=True, related_name='bookings')
    additional_services = models.ManyToManyField(AdditionalService, blank=True)
    property_size = models.CharField(max_length=30, choices=PropertySizeOption.SIZE_CHOICES)
    is_emergency = models.BooleanField(default=False)

    # user details
    user = models.ForeignKey(User, on_delete=models.SET_NULL, null=True, related_name='bookings')
    
    # schedule details
    schedule = models.ForeignKey(Schedule, on_delete=models.SET_NULL, null=True, related_name='bookings')
    service_date = models.DateField()
    time_slot = models.ForeignKey(TimeSlot, on_delete=models.SET_NULL, null=True, related_name='bookings')
    
    # location
    location = models.ForeignKey(ServiceLocation, on_delete=models.SET_NULL, null=True, related_name='bookings')
    
    # pricing
    base_price = models.DecimalField(max_digits=10, decimal_places=2)
    property_multiplier = models.DecimalField(max_digits=4, decimal_places=2, default=1.0)
    add_ons_total = models.DecimalField(max_digits=10, decimal_places=2, default=0.0)
    emergency_fee = models.DecimalField(max_digits=10, decimal_places=2, default=0.0)
    total_price = models.DecimalField(max_digits=10, decimal_places=2)

    # payment
    PAYMENT_METHODS = [
        ('credit_card', 'Credit Card'),
        ('debit_card', 'Debit Card'),
        ('paypal', 'PayPal'),
        ('bank_transfer', 'Bank Transfer'),
    ]
    payment_method = models.CharField(max_length=20, choices=PAYMENT_METHODS)

    # status
    status = models.CharField(max_length=20, choices=StatusChoice.choices, default=StatusChoice.PENDING)
    booked_at = models.DateTimeField(auto_now_add=True)
    updated_at = models.DateTimeField(auto_now=True)

    # recurringg
    is_recurring = models.BooleanField(default=False)
    recurring_booking = models.ForeignKey('RecurringBooking', on_delete=models.SET_NULL, null=True, blank=True, related_name='bookings')

    def total_price(self):
        total = self.main_service.base_price
        for addon in self.additional_services.all():
            total += addon.price
        return total

    def __str__(self):
        return f"Booking by {self.user.username} for {self.main_service.service_type}"
    
    def calculate_price(self):
        total = self.base_price
        total += self.add_ons_total
        total += self.emergency_fee
        total *= self.property_multiplier
        self.total_price = total
        self.save()
        return self.total_price
    
    def save(self, *args, **kwargs):
        # Calculate add_ons_total

        if self.schedule:
            self.service_date = self.schedule.date
            self.time_slot = self.schedule.slot

        
        # Get property multiplier
        property_size_obj = PropertySizeOption.objects.filter(size_type=self.property_size).first()
        self.property_multiplier = property_size_obj.price_multiplier if property_size_obj else Decimal('1.0')
        
        # Calculate add-ons total if booking exists
        if self.pk:
            self.add_ons_total = sum(addon.price for addon in self.additional_services.all())
        
        # Set emergency fee
        if self.is_emergency:
            emergency_addon = AdditionalService.objects.filter(addon_type='emergency').first()
            self.emergency_fee = emergency_addon.price if emergency_addon else Decimal('10000')
        else:
            self.emergency_fee = Decimal('0')
        
        # Calculate total
        self.total_amount = self.calculate_total()
        
        super().save(*args, **kwargs)
    

class RecurringBooking(models.Model):
    """Model to handle recurring bookings."""
    INTERVAL_CHOICES = [
        ('weekly', 'Weekly'),
        ('bi_weekly', 'Bi-Weekly'),
        ('monthly', 'Monthly'),
        ('quarterly', 'Quarterly'),
    ]
    # Upcoming fields to be added for recurring bookings
    user = models.ForeignKey(User, on_delete=models.CASCADE, related_name='recurring_bookings')

    main_service = models.ForeignKey(Service, on_delete=models.SET_NULL, null=True, related_name='recurring_bookings')
    additional_services = models.ManyToManyField(AdditionalService, blank=True)
    cleaning_option = models.ForeignKey(CleaningServiceOption, on_delete=models.SET_NULL, null=True, blank=True)
    property_size = models.CharField(max_length=30, choices=PropertySizeOption.SIZE_CHOICES)    

    location = models.ForeignKey(ServiceLocation, on_delete=models.SET_NULL, null=True, related_name='recurring_bookings')

    time_slot = models.ForeignKey(TimeSlot, on_delete=models.SET_NULL, null=True, related_name='recurring_bookings')
    start_date = models.DateField()
    interval = models.CharField(max_length=20, choices=INTERVAL_CHOICES)
    end_date = models.DateField(null=True, blank=True)  # Optional end date

    created_at = models.DateTimeField(auto_now_add=True)
    updated_at = models.DateTimeField(auto_now=True)
    is_active = models.BooleanField(default=True)

    def __str__(self):
        return f"Recurring Booking by {self.user.username} - {self.main_service.service_type} every {self.interval}"
